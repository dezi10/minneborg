    <script src="script.js"></script>
    <script>
        // Initialiser EmailJS med din Public Key
        (function() {
            emailjs.init({
                publicKey: "V78SaFZYJYRnB1NWc",
            });
        })();

        // Kontaktskjema håndtering
        document.getElementById('contactForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const formMessage = document.getElementById('formMessage');
            const submitBtn = this.querySelector('button[type="submit"]');
            const lang = localStorage.getItem('language') || 'no';
            
            const loadingMessage = lang === 'no' ? 'Sender melding...' : 'Sending message...';
            const successMessage = lang === 'no' 
                ? 'Takk for din melding! Jeg vil svare deg så snart som mulig.' 
                : 'Thank you for your message! I will respond as soon as possible.';
            const errorMessage = lang === 'no'
                ? 'Det oppstod en feil. Vennligst prøv igjen eller kontakt meg via LinkedIn.'
                : 'An error occurred. Please try again or contact me via LinkedIn.';
            
            // Vis lasting
            submitBtn.disabled = true;
            submitBtn.textContent = loadingMessage;
            formMessage.className = 'form-message';
            formMessage.textContent = '';
            
            // Hent skjemadata
            const templateParams = {
                from_name: document.getElementById('name').value,
                reply_to: document.getElementById('email').value,
                subject: document.getElementById('subject').value,
                message: document.getElementById('message').value
            };
            
            // Send e-post via EmailJS
            emailjs.send('service_p3u90dm', 'template_default', templateParams)
                .then(function(response) {
                    console.log('SUCCESS!', response.status, response.text);
                    formMessage.className = 'form-message success';
                    formMessage.textContent = successMessage;
                    
                    // Reset skjema
                    document.getElementById('contactForm').reset();
                    
                    // Skjul melding etter 5 sekunder
                    setTimeout(() => {
                        formMessage.className = 'form-message';
                        formMessage.textContent = '';
                    }, 5000);
                }, function(error) {
                    console.error('FAILED...', error);
                    formMessage.className = 'form-message error';
                    formMessage.textContent = errorMessage;
                })
                .finally(() => {
                    // Reaktiver knapp
                    submitBtn.disabled = false;
                    const btnText = lang === 'no' ? 'Send Melding' : 'Send Message';
                    submitBtn.textContent = btnText;
                });
        });
    </script>
</body>
</html>
